--Process stock 4-7 for prioritization
CREATE VIEW surv_pv_cst.geo_priorities_202608 AS
SELECT *, 
	--assign top half of 70% of total estimated abundance as H
	--assign bottom half of 70% of total estimated abundance as M
	--assign polyids in bottom 30% of total estimate abundance that haven't been surveyed within the last 10 years as L and have an abundance estimate
	--excludes glacial sites
	CASE WHEN rank <= 0.5 * total_rank AND priority_type = 'Top' THEN 'H' 
		WHEN rank > 0.5 * total_rank AND priority_type = 'Top' THEN 'M' 
		WHEN priority_type = 'Bottom' AND last_surveyed <= '2016-12-31' AND abund_est <> 0 THEN 'L'
		ELSE 'not_selected' END AS priority
FROM(
	SELECT stockid, stockname, polyid, last_surveyed, abund_est, running_abund, total_abund, rank, 
		MAX(rank) OVER (PARTITION BY priority_type) AS total_rank, priority_type
	FROM (
		SELECT stockid, stockname, polyid, last_surveyed, abund_est, running_abund, total_abund, 
			Rank() OVER (ORDER BY abund_est DESC) AS rank, 
			--identify polyids in the top 70% of total estimated abundance
			CASE WHEN running_abund <= 0.7 * total_abund THEN 'Top' ELSE 'Bottom' END AS priority_type
		FROM (
		    SELECT *, 
				--get running abundance and total abundance to identify top 70%
		         SUM(abund_est) OVER (ORDER BY abund_est DESC) as running_abund,
		         SUM(abund_est) OVER () as total_abund			 
		    FROM surv_pv_cst.res_abundance_2023
			INNER JOIN (SELECT polyid, stockid, stockname, glacier_name FROM surv_pv_cst.geo_polys) p USING (polyid)
			INNER JOIN (
				SELECT polyid, MAX(survey_dt)::date AS last_surveyed
				FROM surv_pv_cst.summ_count_by_polyid
				GROUP BY polyid
				) d USING (polyid)
			WHERE stockid = 4
			AND glacier_name IS NULL
		) t
		) r
	ORDER BY rank
) f

DROP VIEW surv_pv_cst.geo_priorities_202608;

CREATE VIEW surv_pv_cst.geo_priorities_202608 AS
 SELECT ROW_NUMBER() OVER( ORDER BY polyid) id, polyid,
    geo_polys.stockid,
    geo_polys.stockname,
    geo_polys.station,
    geo_polys.geom,
    tbl_effort_priority.priority_lku AS priority
   FROM surv_pv_cst.tbl_effort_priority
     JOIN surv_pv_cst.geo_polys USING (polyid)
  WHERE tbl_effort_priority.priority_process_date = '2026-04-13'::date;

FROM surv_pv_cst.geo_priorities_202608
GROUP BY (stockname, priority)
ORDER BY (stockname, priority);
